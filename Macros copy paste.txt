Sub color()
'
' color Macro
Dim n, m, i, j As Integer

 n = Cells(Rows.Count, 1).End(xlUp).Row
 m = Cells(7, Columns.Count).End(xlToLeft).Column
  'i = 3
  a = 3
  'MsgBox m
  For j = 1 To 30
      For i = 3 To m / 2 + 2
          Range(Cells(7, a), Cells(n, a)).Select
          With Selection.Interior
         .Pattern = xlSolid
         .PatternColorIndex = xlAutomatic
         .ThemeColor = xlThemeColorAccent5
         .TintAndShade = 0.399975585192419
         .PatternTintAndShade = 0
          End With
        a = a + 2
        Next i
    ActiveSheet.Index + 1.Select
    Next j
End Sub
